#按监狱统计处罚数量
adjudications_clean %>%
  group_by(establishment) %>%
  summarise(total = sum(count)) %>%
  slice_max(total, n = 10) %>%
  ggplot(aes(x = reorder(establishment, total), y = total)) +
  geom_col(fill = "steelblue") +
  coord_flip() +
  labs(title = "Top 10 Prisons by Adjudications (Q1 2023)",
       x = "Prison",
       y = "Total disciplinary cases")

adjudications_by_offence <- adjudications_clean %>%
  group_by(offence) %>%
  summarise(total = sum(count)) %>%
  slice_max(total, n = 10)

ggplot(adjudications_by_offence, aes(x = reorder(offence, total), y = total)) +
  geom_col(fill = "tomato") +
  coord_flip() +
  labs(
    title = "Top 10 Offence Types in Adjudications",
    x = "Offence Type",
    y = "Total Cases"
  )

adjudications_clean %>%
  group_by(sex) %>%
  summarise(total = sum(count)) %>%
  ggplot(aes(x = sex, y = total, fill = sex)) +
  geom_col() +
  labs(
    title = "Adjudications by Sex (Q1 2023)",
    x = "Sex",
    y = "Total Cases"
  )
